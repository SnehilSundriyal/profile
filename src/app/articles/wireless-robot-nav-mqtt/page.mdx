import mqtt "github.com/eclipse/paho.mqtt.golang"

// configuring MQTT connection to broker
options := mqtt.NewClientOptions()
options.AddBroker("tcp://broker.hivemq.com:1883")
options.SetClientID("web-controller")
options.SetKeepAlive(60 * time.Second)

// create an MQTT client (for this broker)
client := mqtt.NewClient(options)

// connect this client (stm32) to the broker
token := client.Connect()
token.Wait()
if token.Error() != nil {
    log.Fatal("connection attempt failed - ", token.Error())
}

// subscribe to updates sent by STM32
client.Subscribe("robot/status", 0, func(client mqtt.Client, msg mqtt.Message) {
    log.Println("current status - ", msg.Payload())
})

// publish commands to STM32
client.Publish("robot/control", 0, false, "FORWARD")
